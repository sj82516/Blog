/**
 * @param {number[][]} matrix
 * @param {number} target
 * @return {boolean}
 */
var searchMatrix = function(matrix, target) {
    var index = [];
    var m=matrix.length;
    var n=matrix[0].length;
    for(var i=0; i<m; i++){
        index.push(matrix[i][n-1]);
    }
    for(var j=0; j<m; j++){
        if(index[j]>target){
            for(var k=0;k<n;k++){
                if(matrix[j][k]===target){
                    return true;
                }
            }
            return false;
        }else if(index[j]===target){
            return true;
        }
    }
    return false;
};
